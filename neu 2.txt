  function showScreen(name) {

  async function callClaude(prompt) {
    const res = await fetch("https://api.anthropic.com/v1/messages", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "x-api-key": API_KEY,
        "anthropic-version": "2023-06-01",
        "anthropic-dangerous-direct-browser-access": "true"
      },
      body: JSON.stringify({
        model: "claude-sonnet-4-20250514",
        max_tokens: 1500,
        messages: [{ role: "user", content: prompt }]
      })
    });
    if (!res.ok) {
      const err = await res.json();
      throw new Error(err.error?.message || "API-Fehler");
    }
    const data = await res.json();
    return data.content[0].text;
  }

  async function runCreate() {
    const sport    = document.getElementById('create-sport').value;
    const players  = document.getElementById('create-players').value;
    const strengths = document.getElementById('create-strengths').value;
    const context  = document.getElementById('create-context').value;
    const style    = document.getElementById('create-style').value;

    const prompt = `Du bist ein erfahrener Taktiktrainer für ${sport}.
Erstelle eine detaillierte Taktikaufstellung für ${players} Spieler.

Stärken des Teams: ${strengths || "nicht angegeben"}
Spielsituation / Gegner: ${context || "nicht angegeben"}
Bevorzugter Spielstil: ${style}

Antworte auf Deutsch mit:
1. Empfohlene Formation
2. Positionsbeschreibung für jede Position
3. Taktische Anweisungen (Offensive & Defensive)
4. Wichtigste Schlüsselprinzipien
5. Kurze Begründung warum diese Taktik passt`;

    await runAI(prompt, 'create');
  }

  async function runRate() {
    const plan   = document.getElementById('rate-plan').value;
    const sport  = document.getElementById('rate-sport').value;
    const level  = document.getElementById('rate-level').value;
    const focus  = document.getElementById('rate-focus').value;

    if (!plan.trim()) {
      alert("Bitte beschreibe zuerst deine Aufstellung.");
      return;
    }

    const prompt = `Du bist ein erfahrener Taktikexperte für ${sport} auf ${level}-Niveau.
Analysiere folgende Aufstellung / Taktik:

${plan}

${focus ? "Besonderer Fokus auf: " + focus : ""}

Antworte auf Deutsch mit einer strukturierten Analyse:
1. Gesamtbewertung (1–10) mit Begründung
2. Stärken der Aufstellung
3. Schwächen / Risiken
4. Konkrete Verbesserungsvorschläge
5. Alternative Formationen die besser passen könnten`;

    await runAI(prompt, 'rate');
  }

  async function runAI(prompt, mode) {
    const btn    = document.getElementById(mode + '-btn');
    const box    = document.getElementById(mode + '-result');
    const output = document.getElementById(mode + '-result-text');

    btn.disabled = true;
    btn.innerHTML = '<span class="spinner"></span> Analysiere...';
    box.classList.add('visible');
    output.textContent = "Die KI denkt nach...";

    try {
      const text = await callClaude(prompt);
      output.textContent = text;
    } catch (e) {
      output.textContent = "Fehler: " + e.message + "\n\nBitte prüfe deinen API-Key.";
    } finally {
      btn.disabled = false;
      btn.innerHTML = mode === 'create' ? 'Aufstellung generieren →' : 'Plan analysieren →';
    }
  }
